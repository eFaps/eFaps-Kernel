21  package org.efaps.bpm.timer;
22  
23  import java.util.Collection;
24  import java.util.Map;
25  import java.util.concurrent.ConcurrentHashMap;
26  
27  import org.drools.core.command.CommandService;
28  import org.drools.core.time.InternalSchedulerService;
29  import org.drools.core.time.Job;
30  import org.drools.core.time.JobContext;
31  import org.drools.core.time.JobHandle;
32  import org.drools.core.time.SelfRemovalJob;
33  import org.drools.core.time.SelfRemovalJobContext;
34  import org.drools.core.time.Trigger;
35  import org.drools.core.time.impl.TimerJobFactoryManager;
36  import org.drools.core.time.impl.TimerJobInstance;
37  
38  
39  
40  /**
41   * A Timer factory that opens a Context.
42   *
43   * @author The eFaps Team
44   * @version $Id$
45   */
46  public class ContextTimerJobFactoryManager
47      implements TimerJobFactoryManager
48  {
49      /**
50       * The CommandSerice that will be executed on time event.
51       */
52      private CommandService commandService;
53  
54      /**
55       * Timer instances.
56       */
57      private final Map<Long, TimerJobInstance> timerInstances;
58  
59      /**
60       * Default Constructor.
61       */
62      public ContextTimerJobFactoryManager()
63      {
64          this.timerInstances = new ConcurrentHashMap<Long, TimerJobInstance>();
65      }
66  
67      /**
68       * Get the Command Service.
69       * @return value of instance variable {@link #commandService}
70       * @see org.drools.time.impl.TimerJobFactoryManager#getCommandService()
71       */
72      public CommandService getCommandService()
73      {
74          return this.commandService;
75      }
76  
77      /**
78       * Set the Command Service.
79       * @param _commandService   Command Service to set
80       * @see org.drools.time.impl.TimerJobFactoryManager#setCommandService(org.drools.command.CommandService)
81       */
82      public void setCommandService(final CommandService _commandService)
83      {
84          this.commandService = _commandService;
85      }
86  
87      /**
88       * @param _job          job to be used for timer
89       * @param _ctx          jobContext
90       * @param _trigger      trigger to use
91       * @param _handle       handle to the job
92       * @param _scheduler    service
93       * @return new ContextTimerJobInstance
94       */
95      public TimerJobInstance createTimerJobInstance(final Job _job,
96                                                     final JobContext _ctx,
97                                                     final Trigger _trigger,
98                                                     final JobHandle _handle,
99                                                     final InternalSchedulerService _scheduler)
100     {
101         _ctx.setJobHandle(_handle);
102         final ContextTimerJobInstance jobInstance = new ContextTimerJobInstance(new SelfRemovalJob(_job),
103                         new SelfRemovalJobContext(_ctx, this.timerInstances),
104                         _trigger,
105                         _handle,
106                         _scheduler);
107         return jobInstance;
108     }
109 
110     /**
111      * @param _instance  instance of a timer job to add
112      * @see org.drools.time.impl.TimerJobFactoryManager#addTimerJobInstance(org.drools.time.impl.TimerJobInstance)
113      */
114     public void addTimerJobInstance(final TimerJobInstance _instance)
115     {
116         this.timerInstances.put(_instance.getJobHandle().getId(),
117                         _instance);
118     }
119 
120     /**
121      * @param _instance  instance of a timer job to remove
122      * @see org.drools.time.impl.TimerJobFactoryManager#removeTimerJobInstance(org.drools.time.impl.TimerJobInstance)
123      */
124     public void removeTimerJobInstance(final TimerJobInstance _instance)
125     {
126         this.timerInstances.remove(_instance.getJobHandle().getId());
127     }
128 
129     /**
130      * @return a collection of TimerJobInstance
131      * @see org.drools.time.impl.TimerJobFactoryManager#getTimerJobInstances()
132      */
133     public Collection<TimerJobInstance> getTimerJobInstances()
134     {
135         return this.timerInstances.values();
136     }
137 }
