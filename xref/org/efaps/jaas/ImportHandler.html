38  /**
39   * The class is used to import from all JAAS system the user into eFaps so that
40   * all users which could authentificate them self outside eFaps are known inside
41   * eFaps (e.g. to assign them for access or to send then an email).<br/> To
42   * start an import, the class must be instantiated and method
43   * {@link #importPersons} must be called.
44   *
45   * @author The eFaps Team
46   * @version $Id$
47   */
48  public class ImportHandler
49      extends LoginHandler
50  {
51      /**
52       * Logging instance used to give logging information of this class.
53       */
54      private static final Logger LOG = LoggerFactory.getLogger(ImportHandler.class);
55  
56      /**
57       * All person mapper instances are stored in this instance variable.
58       */
59      private final Set<PersonMapper> persMappers = new HashSet<PersonMapper>();
60  
61      /**
62       * Map the person mapper instances to the user name of the persons.
63       */
64      private final Map<String, PersonMapper> name2persMapper = new HashMap<String, PersonMapper>();
65  
66      /**
67       * Map the person mapper instances to the eFaps person instances.
68       */
69      private final Map<Person, PersonMapper> pers2persMapper = new HashMap<Person, PersonMapper>();
70  
71      /**
72       * Constructor to initialize the login handler. If <i>null</i> is given to
73       * the application name, the default value defined in {@link #application}
74       * is used.
75       *
76       * @param _application  application name of the JAAS configuration
77       */
78      public ImportHandler(final String _application)
79      {
80          super(_application);
81      }
82  
83      /**
84       * The method calls first method {@link #readPersons} to read all persons
85       * in the cache. Then all found persons are updated with the information
86       * from.
87       *
88       * @see #readPersons()
89       * @see #updatePersons()
90       */
91      public void importPersons()
92      {
93          readPersons();
94          updatePersons();
95      }
96  
97      /**
98       * The algorithm for identification of persons with specified key and name in
99       * a JAAS system is:
100      * <ul>
101      * <li>search person by key in eFaps</li>
102      * <li>if not found search person by name in eFaps</li>
103      * </ul>
104      * If no person inside eFaps is found, search if another JAAS system
105      * identifies a person with the same name!<br/> If an exception is thrown
106      * inside called methods, the exceptions are only written to the log as
107      * errors.
108      */
109     protected void readPersons()
110     {
111         try {
112             final LoginContext login =  new LoginContext(
113                     getApplicationName(),
114                     new LoginCallbackHandler(ActionCallback.Mode.ALL_PERSONS, null, null));
115             login.login();
116             for (final JAASSystem system : JAASSystem.getAllJAASSystems()) {
117                 if (ImportHandler.LOG.isDebugEnabled()) {
118                     ImportHandler.LOG.debug("check JAAS system '" + system + "'");
119                 }
120                 final Set<?> users = login.getSubject().getPrincipals(system.getPersonJAASPrincipleClass());
121                 for (final Object persObj : users) {
122                     if (ImportHandler.LOG.isDebugEnabled()) {
123                         ImportHandler.LOG.debug("- check person '" + persObj + "'");
124                     }
125                     try {
126                         final String persKey = (String) system.getPersonMethodKey().invoke(persObj, (Object) null);
127                         final String persName = (String) system.getPersonMethodName().invoke(persObj, (Object) null);
128 
129                         PersonMapper persMapper = null;
130 
131                         Person foundPerson = Person.getWithJAASKey(system, persKey);
132                         if (foundPerson == null) {
133                             foundPerson = Person.get(persName);
134                         }
135 
136                         if (foundPerson == null) {
137                             persMapper = this.name2persMapper.get(persName);
138                         } else {
139                             persMapper = this.pers2persMapper.get(foundPerson);
140                             if (persMapper == null) {
141                                 persMapper = this.name2persMapper.get(persName);
142                             }
143                         }
144 
145                         if (persMapper == null) {
146                             persMapper = new PersonMapper(foundPerson, persName);
147                         }
148 
149                         persMapper.addJAASSystem(system, persKey);
150 
151                     } catch (final EFapsException e) {
152                         ImportHandler.LOG.error("could not search for person with JAAS key " + system.getName(), e);
153                     } catch (final IllegalAccessException e) {
154                         ImportHandler.LOG.error("could not execute person key method for system "
155                                 + system.getName(), e);
156                     } catch (final IllegalArgumentException e) {
157                         ImportHandler.LOG.error("could not execute person key method for system "
158                                 + system.getName(), e);
159                     } catch (final InvocationTargetException e) {
160                         ImportHandler.LOG.error("could not execute person key method for system "
161                                 + system.getName(), e);
162                     }
163                 }
164             }
165         } catch (final LoginException e) {
166             ImportHandler.LOG.error("could not create login context", e);
167         }
168     }
169 
170     /**
171      * <p>Each found person is updated in eFaps (or created if not already
172      * existing). The methods from the login handler {@link LoginHandler} are
173      * used for the update.</p>
174      * <p>Each user name is written as debug information to the log file.</p>
175      * <p>If an exception is thrown inside called methods, the exceptions
176      * are only written to the log as errors.</p>
177      *
178      * @see LoginHandler#getPerson(LoginContext)
179      * @see LoginHandler#createPerson(LoginContext)
180      * @see LoginHandler#updatePerson(LoginContext, Person)
181      * @see LoginHandler#updateRoles(LoginContext, Person)
182      * @see LoginHandler#updateGroups(LoginContext, Person)
183      */
184     protected void updatePersons()
185     {
186         for (final PersonMapper persMapper : this.persMappers) {
187             if (ImportHandler.LOG.isDebugEnabled()) {
188                 ImportHandler.LOG.debug("update person '" + persMapper.name + "'");
189             }
190             try {
191                 final LoginContext login =
192                     new LoginContext(getApplicationName(), new LoginCallbackHandler(
193                             ActionCallback.Mode.PERSON_INFORMATION, persMapper.name, null));
194                 login.login();
195 
196                 Person person = getPerson(login);
197 
198                 if (person == null) {
199                     person = createPerson(login);
200                 }
201 
202                 if (person != null) {
203 
204                     updatePerson(login, person);
205 
206                     person.cleanUp();
207 
208                     updateRoles(login, person);
209                     updateGroups(login, person);
210                 }
211             } catch (final EFapsException e) {
212                 ImportHandler.LOG.error("update failed for '" + persMapper.name + "'", e);
213             } catch (final LoginException e) {
214                 ImportHandler.LOG.error("update failed for '" + persMapper.name + "'", e);
215             }
216         }
217     }
218 
219     /**
220      * Person mapper class.
221      */
222     private final class PersonMapper
223     {
224         /**
225          * eFaps Person instance.
226          */
227         private final Person person;
228 
229         /**
230          * User Name of the person.
231          */
232         private final String name;
233 
234         /**
235          * Mapping of the key for the keys.
236          */
237         private final Map<JAASSystem, String> keys = new HashMap<JAASSystem, String>();
238 
239         /**
240          * If a new person mapper object is instantiated, the mapping for names
241          * to person mapper in {@link #name2persMapper} and the mapping for
242          * persons to person mapper in {@link #pers2persMapper} are updated.
243          * The new instance is added to the list of all person mappers in
244          * {@link #persMappers}.
245          *
246          * @param _person   JAVA person instance (if already existing in eFaps)
247          * @param _name     name of the person
248          */
249         private PersonMapper(final Person _person,
250                              final String _name)
251         {
252             this.person = _person;
253             this.name = _name;
254             ImportHandler.this.name2persMapper.put(this.name, this);
255             if (this.person != null) {
256                 ImportHandler.this.pers2persMapper.put(this.person, this);
257             }
258             ImportHandler.this.persMappers.add(this);
259         }
260 
261         /**
262          * A new key of the person for the JAAS system is defined for the
263          * person.
264          *
265          * @param _jaasSystem   JAAS system in which the person is defined
266          * @param _key          key of the person in the given JAAS system
267          */
268         private void addJAASSystem(final JAASSystem _jaasSystem,
269                                    final String _key)
270         {
271             this.keys.put(_jaasSystem, _key);
272         }
273     }
