21  package org.efaps.update.schema.program.esjp;
22  
23  import java.net.URL;
24  import java.util.UUID;
25  import java.util.regex.Matcher;
26  import java.util.regex.Pattern;
27  
28  import org.efaps.ci.CIAdminProgram;
29  import org.efaps.update.schema.program.AbstractSourceImporter;
30  import org.efaps.update.util.InstallationException;
31  
32  /**
33   * Java source code could be imported into eFaps as ESJP with this class. The
34   * class does not need any XML update files and could be called directly.
35   *
36   * @author The eFaps Team
37   * @version $Id$
38   * TODO:  encoding from java files!
39   */
40  public class ESJPImporter
41      extends AbstractSourceImporter
42  {
43      /**
44       * Constructor used to read the source code from given URL and extract the
45       * class name.
46       *
47       * @param _url    url to the ESJP source code
48       * @throws InstallationException on error
49       */
50      public ESJPImporter(final URL _url)
51          throws InstallationException
52      {
53          super(CIAdminProgram.Java, _url);
54      }
55  
56      /**
57       * This Method extracts the package name and sets the name of this Java
58       * definition (the name is the package name together with the name of the
59       * file excluding the <code>.java</code>).
60       *
61       * @return class name of the ESJP
62       */
63      @Override
64      protected String evalProgramName()
65      {
66          final String urlPath = getUrl().getPath();
67          String name = urlPath.substring(urlPath.lastIndexOf('/') + 1);
68          name = name.substring(0, name.lastIndexOf('.'));
69  
70          // regular expression for the package name
71          final Pattern pckPattern = Pattern.compile("package +[^;]+;");
72          final Matcher pckMatcher = pckPattern.matcher(getCode());
73          if (pckMatcher.find()) {
74              final String pkg = pckMatcher.group()
75                                           .replaceFirst("^(package) +", "")
76                                           .replaceFirst(";$", "");
77              name = pkg + "." + name;
78          }
79          return name;
80      }
81  
82      /**
83       * @return check if this esjp is update allowed
84       */
85      public boolean isUpdate()
86      {
87          return !getCode().toString().contains("@EFapsNoUpdate");
88      }
89  
90      /**
91       * This Method extracts the UUID from the ESJP.
92       *
93       * @return UUID of the ESJP
94       */
95      @Override
96      protected UUID evalUUID()
97      {
98          UUID uuid = null;
99          final Pattern uuidPattern = Pattern.compile("@EFapsUUID ?\\( ?\\\"[0-9a-z\\-]*\\\" ?\\)");
100         final Matcher uuidMatcher = uuidPattern.matcher(getCode());
101         if (uuidMatcher.find()) {
102             final String uuidStr = uuidMatcher.group()
103                                               .replaceFirst("^@EFapsUUID ?\\( ?\\\"", "")
104                                               .replaceFirst("\\\" ?\\)", "");
105             uuid = UUID.fromString(uuidStr);
106         }
107         return uuid;
108     }
109 
110     /**
111      * This Method extracts the Revision from the ESJP.
112      *
113      * @return revision of the ESJP
114      */
115     @Override
116     protected String evalRevision()
117     {
118         String ret = null;
119         final Pattern revisionPattern = Pattern.compile("@EFapsRevision ?\\( ?\\\".*\\\" ?\\)");
120         final Matcher revisionMatcher = revisionPattern.matcher(getCode());
121         if (revisionMatcher.find()) {
122             ret = revisionMatcher.group()
123                                  .replaceFirst("^@EFapsRevision ?\\( ?\\\"", "")
124                                  .replaceFirst("\\\" ?\\)", "");
125         }
126         return ret;
127     }
128 }
