21  package org.efaps.update.schema.program;
22  
23  import java.net.URL;
24  import java.util.Set;
25  
26  import org.efaps.db.Instance;
27  import org.efaps.update.AbstractUpdate;
28  import org.efaps.update.UpdateLifecycle;
29  import org.efaps.update.schema.program.esjp.ESJPImporter;
30  import org.efaps.update.util.InstallationException;
31  
32  /**
33   * Handles the import / update of ESJP's (eFaps stored Java Programs) for eFaps
34   * read from a plain Java source code file.
35   *
36   * @author The eFaps Team
37   * @version $Id$
38   */
39  public class JavaUpdate
40      extends AbstractSourceUpdate
41  {
42      /**
43       * Default constructor to initialize this ESJP update instance for given
44       * <code>_url</code>.
45       *
46       * @param _url  url to the file
47       */
48      public JavaUpdate(final URL _url)
49      {
50          super(_url, "Admin_Program_Java");
51      }
52  
53      /**
54       * If the extension of the file is <code>.java</code>, the method returns
55       * an instance of this class. The instance of this class owns one definition
56       * instance where the code and the name is defined.
57       *
58       * @param _url      URL of the file depending of the root URL
59       * @return Java update definition read by digester
60       */
61      public static JavaUpdate readFile(final URL _url)
62      {
63          final JavaUpdate ret = new JavaUpdate(_url);
64          final JavaDefinition definition = ret.new JavaDefinition(_url);
65          ret.addDefinition(definition);
66          return ret;
67      }
68  
69      /**
70       * The Java definition holds the code and the name of the Java class.
71       */
72      public class JavaDefinition
73          extends AbstractSourceDefinition
74      {
75          /**
76           * Importer for the ESJP.
77           */
78          private ESJPImporter importer = null;
79  
80          /**
81           * Is update of the given esjp allowed.
82           */
83          private boolean updateAllowed = true;
84  
85          /**
86           * @param _url URL to the java file
87           */
88          protected JavaDefinition(final URL _url)
89          {
90              super(_url);
91          }
92  
93          /**
94           * Search the instance.
95           *
96           * @throws InstallationException if the Java source code could not be
97           *                               read or the file could not be accessed
98           *                               because of the wrong URL
99           */
100         @Override
101         protected void searchInstance()
102             throws InstallationException
103         {
104             if (this.importer == null) {
105                 this.importer = new ESJPImporter(getUrl());
106             }
107             setName(this.importer.getProgramName());
108 
109             if (this.importer.getEFapsUUID() != null) {
110                 addValue("UUID", this.importer.getEFapsUUID().toString());
111             }
112 
113             if (this.importer.getRevision() != null) {
114                 addValue("Revision", this.importer.getRevision());
115             }
116             if (getInstance() == null) {
117                 final Instance instTmp = this.importer.searchInstance();
118                 setInstance(instTmp);
119                 if (instTmp == null) {
120                     this.updateAllowed = true;
121                 } else {
122                     this.updateAllowed = this.importer.isUpdate();
123                 }
124             }
125         }
126 
127         /**
128          * The method overwrites the method from the super class, because a
129          * check in of the Java source code is needed after the update in the
130          * database.
131          *
132          * @param _step             current update step
133          * @param _allLinkTypes     all link types to update
134          * @throws InstallationException on error
135          */
136         @Override
137         public void updateInDB(final UpdateLifecycle _step,
138                                final Set<Link> _allLinkTypes)
139             throws InstallationException
140         {
141             if (_step == UpdateLifecycle.EFAPS_UPDATE)  {
142                 if (this.updateAllowed) {
143                     AbstractUpdate.LOG.info("    Update {} '{}'", getInstance().getType().getName(),
144                                     this.importer.getProgramName());
145                     this.importer.updateDB(getInstance());
146                 } else {
147                     AbstractUpdate.LOG.info("    No Update set for esjp: {}", this.importer.getProgramName());
148                 }
149             } else  {
150                 super.updateInDB(_step, _allLinkTypes);
151             }
152         }
153 
154         /**
155          * {@inheritDoc}
156          */
157         @Override
158         protected String getRevision()
159             throws InstallationException
160         {
161             if (this.importer == null) {
162                 this.importer = new ESJPImporter(getUrl());
163             }
164             return this.importer.getRevision();
165         }
166     }
167 }
